function doPost(e) {
  let jsonString = e.postData.getDataAsString();
  let data = JSON.parse(jsonString);

  let ss = SpreadsheetApp.openById('スプレッドシートのID');
  let sheet = ss.getSheetByName("シート1");

  sheet.appendRow([data.temperature,data.humidity]);
  return ContentService.createTextOutput(JSON.stringify(e));
}
